SELECT
  ID,
  firstName as FIRST_NAME,
  lastName as LAST_NAME,
  dateOfBirth,
  telephone,
  address,
  city,
  postal_code,
  provinceID,
  email,
  medicareNumber,
  CASE WHEN CS.isCitizen = 1 THEN
	'YES'
  ELSE
	'NO'
  END IS_CITIZEN,
  CS.IDNumber
FROM
  Employee P
LEFT JOIN CitizenshipStatus CS ON CS.patientID = P.ID
WHERE ID = ?
